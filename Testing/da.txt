class Event2(models.Model):
    venue = models.CharField(max_length=10)
    dateWtime_start = models.DateTimeField(auto_now_add=False)
    dateWtime_end = models.DateTimeField(auto_now_add=False)
    event_occured = models.CharField(max_length=10)
    instructor = models.CharField(max_length=30)

    def __str__(self):
        return f'{self.venue}'
